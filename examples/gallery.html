Hello MathViz

Beginner

MathViz Code

scene HelloMathViz {
    fn construct(self) {
        let title = Text("Hello, MathViz!")
        title.scale(1.5)

        play(Write(title))
        wait(1.0)
        play(title.animate.set_color(BLUE))
        wait(0.5)
        play(FadeOut(title))
    }
}

Circle Area Formula

Math

MathViz Code

scene CircleArea {
    fn construct(self) {
        let circle = Circle(radius: 2.0, color: BLUE)
        let formula = MathTex("A = \\pi r^2")
        formula.next_to(circle, DOWN)

        play(Create(circle))
        play(Write(formula))
        wait(1.0)
        play(circle.animate.set_fill(BLUE, opacity: 0.5))
        wait(0.5)
        play(FadeOut(circle, formula))
    }
}

Shape Transformation

Animation

MathViz Code

scene ShapeTransform {
    fn construct(self) {
        let square = Square(side_length: 2.0, color: RED)
        let circle = Circle(radius: 1.0, color: BLUE)

        play(Create(square))
        wait(0.5)
        play(Transform(square, circle))
        wait(0.5)
        play(circle.animate.scale(2.0))
        wait(0.5)
        play(FadeOut(circle))
    }
}

Function Graph

Advanced

MathViz Code

scene FunctionGraph {
    fn construct(self) {
        let axes = Axes(
            x_range: [-3, 3, 1],
            y_range: [-2, 2, 1],
            axis_config: {"color": BLUE}
        )

        let graph = axes.plot(|x| sin(x), color: YELLOW)
        let label = MathTex("f(x) = \\sin(x)")
        label.to_corner(UR)

        play(Create(axes))
        play(Create(graph), Write(label))
        wait(1.0)
        play(FadeOut(axes, graph, label))
    }
}
