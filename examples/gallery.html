Getting Started - Hello World

What you will learn

  • The scene keyword defines a Manim Scene class
  • fn construct(self) is the entry point for every animation
  • self.play() runs an animation and self.wait() pauses
  • Write and FadeOut are basic animation types
// Tutorial 1: Getting Started - Hello World
// Your first MathViz animation

scene HelloWorld {
    fn construct(self) {
        let title = Text("Hello, MathViz!")
        title.scale(1.5)

        self.play(Write(title))
        self.wait(1.0)
        self.play(title.animate.set_color(BLUE))
        self.wait(0.5)
        self.play(FadeOut(title))
    }
}

Basic Animations

What you will learn

  • from manim import brings specific Manim objects into scope
  • Create() draws a shape stroke by stroke
  • FadeIn() and FadeOut() fade objects in and out
  • Transform(a, b) morphs object a into shape b
  • set_fill(color, opacity) fills the interior of a shape
// Tutorial 2: Basic Animations
// Create, FadeIn, FadeOut, Transform

from manim import Circle, Square, Create, FadeIn, FadeOut, Transform, BLUE, RED

scene BasicAnimations {
    fn construct(self) {
        let circle = Circle()
        circle.set_fill(BLUE, 0.5)

        // Create animation draws the shape
        self.play(Create(circle))
        self.wait(0.5)

        // FadeOut makes objects disappear
        self.play(FadeOut(circle))
        self.wait(0.5)

        // FadeIn makes objects appear
        let square = Square()
        square.set_fill(RED, 0.5)
        self.play(FadeIn(square))
        self.wait(0.5)

        // Transform morphs one shape into another
        let circle2 = Circle()
        circle2.set_fill(BLUE, 0.5)
        self.play(Transform(square, circle2))
        self.wait(1.0)
    }
}

Working with Shapes

What you will learn

  • MathViz uses colon syntax for named parameters: radius: 1.0 instead of radius=1.0
  • Circle, Square, Triangle, Line, and Arrow are core shapes
  • shift() moves objects by a direction vector (e.g., LEFT * 3)
  • Multiple animations can play simultaneously when passed to the same self.play()
// Tutorial 3: Working with Shapes
// Circle, Square, Triangle, Line, Arrow

from manim import Circle, Square, Triangle, Line, Arrow
from manim import Create, FadeOut, BLUE, RED, GREEN, YELLOW, LEFT, RIGHT, UP, DOWN

scene WorkingWithShapes {
    fn construct(self) {
        // Circle with radius
        let circle = Circle(radius: 1.0, color: BLUE)
        circle.shift(LEFT * 3)

        // Square with side length
        let square = Square(side_length: 2.0, color: RED)

        // Triangle
        let tri = Triangle(color: GREEN)
        tri.shift(RIGHT * 3)

        self.play(Create(circle), Create(square), Create(tri))
        self.wait(1.0)

        // Line between two points
        let line = Line(LEFT * 2, RIGHT * 2, color: YELLOW)
        line.shift(DOWN * 2)
        self.play(Create(line))
        self.wait(1.0)

        self.play(FadeOut(circle), FadeOut(square), FadeOut(tri), FadeOut(line))
    }
}

Text and LaTeX

What you will learn

  • Text("...") creates styled text objects
  • MathTex("...") renders LaTeX mathematical formulas
  • Write() is the animation for text and formulas
  • next_to(obj, direction, buff: spacing) positions relative to another object
  • Use double backslashes \\ for LaTeX commands in strings
// Tutorial 4: Text and LaTeX
// Text objects, MathTex, Tex

scene TextAndLatex {
    fn construct(self) {
        // Simple text
        let title = Text("The Pythagorean Theorem", color: BLUE)
        title.scale(0.8)
        title.to_edge(UP)
        self.play(Write(title))

        // LaTeX mathematical formula
        let formula = MathTex("a^2 + b^2 = c^2")
        formula.scale(1.5)
        self.play(Write(formula))
        self.wait(1.0)

        // More complex LaTeX
        let integral = MathTex("\\int_0^1 x^2 \\, dx = \\frac{1}{3}")
        integral.next_to(formula, DOWN, buff: 0.5)
        self.play(Write(integral))
        self.wait(1.0)

        self.play(FadeOut(title), FadeOut(formula), FadeOut(integral))
    }
}

Positioning and Layout

What you will learn

  • shift(direction * amount) moves an object by a vector offset
  • to_edge(UP) places an object at the screen edge
  • to_corner(UL) places an object in a screen corner
  • next_to(other, direction, buff: spacing) positions relative to another object
  • Direction constants: UP, DOWN, LEFT, RIGHT, UL, UR, etc.
// Tutorial 5: Positioning and Layout
// next_to, shift, to_edge, to_corner

from manim import Circle, Square, Text, Create, Write, FadeOut
from manim import BLUE, RED, GREEN, LEFT, RIGHT, UP, DOWN, UL, UR

scene PositioningLayout {
    fn construct(self) {
        // Center object
        let center_sq = Square(side_length: 1.0, color: RED)
        self.play(Create(center_sq))
        self.wait(0.5)

        // Place object next_to another
        let label = Text("Center", color: BLUE)
        label.scale(0.6)
        label.next_to(center_sq, DOWN, buff: 0.3)
        self.play(Write(label))

        // shift moves objects by a vector
        let circle = Circle(radius: 0.5, color: GREEN)
        circle.shift(RIGHT * 3)
        self.play(Create(circle))

        // to_edge places at screen edge
        let top_text = Text("Top Edge", color: BLUE)
        top_text.scale(0.5)
        top_text.to_edge(UP)
        self.play(Write(top_text))

        // to_corner places at screen corner
        let corner_text = Text("Corner", color: GREEN)
        corner_text.scale(0.5)
        corner_text.to_corner(UL)
        self.play(Write(corner_text))

        self.wait(1.0)
        self.play(
            FadeOut(center_sq), FadeOut(label),
            FadeOut(circle), FadeOut(top_text), FadeOut(corner_text)
        )
    }
}

Colors and Styling

What you will learn

  • set_color(COLOR) changes the overall color of a shape
  • set_fill(color, opacity: value) fills the interior
  • set_stroke(color, width: value) styles the border
  • Color changes can be animated with .animate.set_color()
  • Available colors: BLUE, RED, GREEN, YELLOW, WHITE, and more
// Tutorial 6: Colors and Styling
// set_color, set_fill, set_stroke

from manim import Circle, Square, Create, FadeOut
from manim import BLUE, RED, GREEN, YELLOW, WHITE, LEFT, RIGHT

scene ColorsStyling {
    fn construct(self) {
        // set_color changes the overall color
        let c1 = Circle(radius: 1.0)
        c1.set_color(BLUE)
        c1.shift(LEFT * 3)

        // set_fill adds interior color with opacity
        let c2 = Circle(radius: 1.0)
        c2.set_fill(RED, opacity: 0.5)
        c2.set_stroke(WHITE, width: 2)

        // Combine fill and stroke
        let c3 = Circle(radius: 1.0)
        c3.set_fill(GREEN, opacity: 0.7)
        c3.set_stroke(YELLOW, width: 4)
        c3.shift(RIGHT * 3)

        self.play(Create(c1), Create(c2), Create(c3))
        self.wait(1.0)

        // Animate color change
        self.play(c1.animate.set_color(YELLOW))
        self.play(c2.animate.set_fill(BLUE, opacity: 0.8))
        self.wait(1.0)

        self.play(FadeOut(c1), FadeOut(c2), FadeOut(c3))
    }
}

Transforms and Morphing

What you will learn

  • Transform(a, b) morphs a into the shape of b (keeps reference to a)
  • ReplacementTransform(a, b) replaces a with b entirely
  • .animate allows chaining multiple property changes in one animation
  • Chain methods: obj.animate.scale(0.5).shift(UP * 2)
  • rotate(PI / 4) rotates an object by an angle in radians
// Tutorial 7: Transforms and Morphing
// Transform, ReplacementTransform, animate

from manim import Circle, Square, Triangle, Transform, ReplacementTransform
from manim import Create, FadeOut, BLUE, RED, GREEN

scene TransformsMorphing {
    fn construct(self) {
        let square = Square(side_length: 2.0, color: BLUE)
        self.play(Create(square))
        self.wait(0.5)

        // Transform morphs one shape into another
        let circle = Circle(radius: 1.0, color: RED)
        self.play(Transform(square, circle))
        self.wait(0.5)

        // ReplacementTransform replaces the mobject reference
        let triangle = Triangle(color: GREEN)
        triangle.scale(2.0)
        self.play(ReplacementTransform(circle, triangle))
        self.wait(0.5)

        // animate allows chaining transformations
        self.play(triangle.animate.scale(0.5).shift(UP * 2))
        self.wait(0.5)

        self.play(triangle.animate.rotate(PI / 4))
        self.wait(1.0)

        self.play(FadeOut(triangle))
    }
}

Graphs and Functions

What you will learn

  • Axes(x_range: [...], y_range: [...]) creates coordinate axes
  • axes.plot(|x| sin(x), color: YELLOW) plots a function using pipe-lambda syntax
  • Pipe-lambda |x| expr compiles to Python lambda x: np.expr(x)
  • MathTex() creates labels for formulas
  • to_corner(UR) places labels in the upper-right corner
// Tutorial 8: Graphs and Functions
// Axes, NumberPlane, plotting

from manim import Axes, Create, Write, FadeOut, BLUE, YELLOW, UR

scene GraphsFunctions {
    fn construct(self) {
        // Create coordinate axes
        let axes = Axes(
            x_range: [-3, 3, 1],
            y_range: [-2, 2, 1]
        )

        // Plot a sine function
        let graph = axes.plot(|x| sin(x), color: YELLOW)
        let label = MathTex("f(x) = \\sin(x)")
        label.to_corner(UR)

        self.play(Create(axes))
        self.play(Create(graph), Write(label))
        self.wait(1.0)

        // Plot a second function
        let graph2 = axes.plot(|x| cos(x), color: BLUE)
        let label2 = MathTex("g(x) = \\cos(x)", color: BLUE)
        label2.next_to(label, DOWN)

        self.play(Create(graph2), Write(label2))
        self.wait(1.0)

        self.play(FadeOut(axes), FadeOut(graph), FadeOut(graph2), FadeOut(label), FadeOut(label2))
    }
}
import numpy as np

Groups and VGroups

What you will learn

  • VGroup(obj1, obj2, obj3) groups multiple mobjects together
  • group.arrange(RIGHT, buff: 0.5) arranges objects in a direction with spacing
  • Animating the group affects all children: group.animate.scale(1.5)
  • Individual elements remain accessible: c1.animate.set_color(YELLOW)
// Tutorial 9: Groups and VGroups
// Grouping objects, batch operations

from manim import Circle, Square, VGroup, Create, FadeOut
from manim import BLUE, RED, GREEN, LEFT, RIGHT

scene GroupsVGroups {
    fn construct(self) {
        // Create individual shapes
        let c1 = Circle(radius: 0.5, color: RED)
        let c2 = Circle(radius: 0.5, color: GREEN)
        let c3 = Circle(radius: 0.5, color: BLUE)

        // Create a VGroup and arrange them
        let group = VGroup(c1, c2, c3)
        group.arrange(RIGHT, buff: 0.5)

        self.play(Create(group))
        self.wait(0.5)

        // Animate the entire group at once
        self.play(group.animate.scale(1.5))
        self.wait(0.5)

        self.play(group.animate.shift(UP * 2))
        self.wait(0.5)

        // Animate individual elements in the group
        self.play(c1.animate.set_color(YELLOW))
        self.wait(1.0)

        self.play(FadeOut(group))
    }
}

Camera and Scene Control

What you will learn

  • run_time: seconds controls the duration of an animation
  • Successive self.play() calls create sequential animations
  • self.wait(seconds) adds pauses between animations
  • Combine set_fill with Create for filled shapes
  • Chain shift() calls for back-and-forth motion patterns
// Tutorial 10: Camera and Scene Control
// Scene configuration, background color, wait times

from manim import Circle, Square, Text, Create, Write, FadeOut
from manim import BLUE, RED, WHITE, UP

scene CameraSceneControl {
    fn construct(self) {
        // Create a complex scene
        let title = Text("Scene Control", color: BLUE)
        title.to_edge(UP)
        self.play(Write(title))

        let circle = Circle(radius: 1.5, color: RED)
        circle.set_fill(RED, opacity: 0.3)
        self.play(Create(circle))

        // Animate with run_time parameter for speed control
        self.play(circle.animate.scale(2.0), run_time: 2.0)
        self.wait(0.5)

        self.play(circle.animate.scale(0.5), run_time: 0.5)
        self.wait(0.5)

        // Successive transformations
        self.play(circle.animate.shift(LEFT * 2), run_time: 1.0)
        self.play(circle.animate.shift(RIGHT * 4), run_time: 1.0)
        self.play(circle.animate.shift(LEFT * 2), run_time: 1.0)

        self.wait(1.0)
        self.play(FadeOut(title), FadeOut(circle))
    }
}

11

Advanced Mathematical Scenes

What you will learn

  • Combine all previous concepts into a complex theorem animation
  • Use MathTex for multi-step proofs and formulas
  • Organize scenes with helper variables and labeled objects
  • Use VGroup to manage complex layouts with multiple elements
  • Apply Transform and ReplacementTransform for proof transitions
// Advanced: Pythagorean Theorem Visualization
// Combines shapes, text, transforms, and layout

from manim import Square, Text, MathTex, VGroup, Create, Write
from manim import FadeIn, FadeOut, Transform, BLUE, RED, GREEN, YELLOW
from manim import UP, DOWN, LEFT, RIGHT

scene PythagoreanTheorem {
    fn construct(self) {
        // Title
        let title = Text("Pythagorean Theorem", color: BLUE)
        title.scale(0.9)
        title.to_edge(UP)
        self.play(Write(title))

        // Show the formula
        let formula = MathTex("a^2 + b^2 = c^2")
        formula.scale(1.5)
        self.play(Write(formula))
        self.wait(1.0)

        // Create visual squares
        let sq_a = Square(side_length: 1.0, color: RED)
        sq_a.set_fill(RED, opacity: 0.3)
        let label_a = MathTex("a^2", color: RED)
        label_a.scale(0.7)

        let sq_b = Square(side_length: 1.5, color: GREEN)
        sq_b.set_fill(GREEN, opacity: 0.3)
        let label_b = MathTex("b^2", color: GREEN)
        label_b.scale(0.7)

        let sq_c = Square(side_length: 2.0, color: YELLOW)
        sq_c.set_fill(YELLOW, opacity: 0.3)
        let label_c = MathTex("c^2", color: YELLOW)
        label_c.scale(0.7)

        // Position squares
        sq_a.shift(LEFT * 3 + DOWN * 1)
        label_a.next_to(sq_a, DOWN)
        sq_b.shift(DOWN * 1)
        label_b.next_to(sq_b, DOWN)
        sq_c.shift(RIGHT * 3 + DOWN * 1)
        label_c.next_to(sq_c, DOWN)

        // Animate the visualization
        self.play(FadeOut(formula))
        self.play(Create(sq_a), Write(label_a))
        self.play(Create(sq_b), Write(label_b))
        self.play(Create(sq_c), Write(label_c))
        self.wait(1.0)

        // Final formula
        let final_eq = MathTex("a^2", "+", "b^2", "=", "c^2")
        final_eq.scale(1.3)
        final_eq.next_to(title, DOWN, buff: 0.5)
        self.play(Write(final_eq))
        self.wait(2.0)

        // Clean up
        let all_objs = VGroup(
            title, sq_a, sq_b, sq_c,
            label_a, label_b, label_c, final_eq
        )
        self.play(FadeOut(all_objs))
    }
}

import numba
from manim import *
from manim import Square, Text, MathTex, VGroup, Create, Write
from manim import FadeIn, FadeOut, Transform, BLUE, RED, GREEN, YELLOW
from manim import UP, DOWN, LEFT, RIGHT

class PythagoreanTheorem(Scene):
    def construct(self):
        title = Text('Pythagorean Theorem', color=BLUE)
        title.scale(0.9)
        title.to_edge(UP)
        self.play(Write(title))
        formula = MathTex('a^2 + b^2 = c^2')
        formula.scale(1.5)
        self.play(Write(formula))
        self.wait(1.0)
        sq_a = Square(side_length=1.0, color=RED)
        sq_a.set_fill(RED, opacity=0.3)
        label_a = MathTex('a^2', color=RED)
        label_a.scale(0.7)
        sq_b = Square(side_length=1.5, color=GREEN)
        sq_b.set_fill(GREEN, opacity=0.3)
        label_b = MathTex('b^2', color=GREEN)
        label_b.scale(0.7)
        sq_c = Square(side_length=2.0, color=YELLOW)
        sq_c.set_fill(YELLOW, opacity=0.3)
        label_c = MathTex('c^2', color=YELLOW)
        label_c.scale(0.7)
        sq_a.shift((LEFT * 3 + DOWN * 1))
        label_a.next_to(sq_a, DOWN)
        sq_b.shift((DOWN * 1))
        label_b.next_to(sq_b, DOWN)
        sq_c.shift((RIGHT * 3 + DOWN * 1))
        label_c.next_to(sq_c, DOWN)
        self.play(FadeOut(formula))
        self.play(Create(sq_a), Write(label_a))
        self.play(Create(sq_b), Write(label_b))
        self.play(Create(sq_c), Write(label_c))
        self.wait(1.0)
        final_eq = MathTex('a^2', '+', 'b^2', '=', 'c^2')
        final_eq.scale(1.3)
        final_eq.next_to(title, DOWN, buff=0.5)
        self.play(Write(final_eq))
        self.wait(2.0)
        all_objs = VGroup(title, sq_a, sq_b, sq_c, label_a, label_b, label_c, final_eq)
        self.play(FadeOut(all_objs))
